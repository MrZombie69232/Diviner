177  subroutine dfindy(xx,yy,n,x, y)
178 c
179 c-----------------------------------------------------------------------
180 c general purpose linear interpolation routine
181 c
182 c inputs: xx x array of values (monotoically increasing)
183 c yy y array of function values
184 c n numer in x and y arrays
185 c x the value of x that we want to know the best
186 c guess value of y for
187 c outputs: y best guess linearly interpolated value
188 c-----------------------------------------------------------------------
189 c
190  real*8 xx(n),yy(n),x,y
191 c
192  if (x.gt.xx(n)) then
193  y=yy(n)

195  y=yy(1)
196  else
197 c find index of xx in x
