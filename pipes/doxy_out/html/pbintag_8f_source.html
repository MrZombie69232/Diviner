4 c PURPOSE: This program adds numerical tags to data based on a user-specified
5 c binning scheme, and the values of the coordinates at the center of the bin
6 c
7 c REQUIRED KEYWORDS:
8 c
9 c x=char -- The first variable you want to index the bins by. The bins
10 c will be indexed as bin(x,y). For example, one could set
11 c set up the bins to be indexed by "lon" and "lat" ( bin(lon,lat) ).
12 c
13 c y=char -- See x=char.
14 c
15 c
16 c xrange=real,real -- The range of x values you want the bins to encompass (xmin and xmax).
17 c (eg lon=0.0,360.0)
18 c
19 c yrange=real,real -- The range of y values you want the bins to encompass (ymin and ymax).
20 c (eg lat=-90.0,90.0)
21 c
22 c nbinx=int -- Number of bins along the x-axis
23 c --OR--
24 c deltax=real -- Increment of the bins along the x-axis
25 c (eg every 2.0 degrees longitude)
26 c
27 c nbiny=int -- Number of bins along the y-axis
28 c --OR--
29 c deltay=real -- Increment of the bins along the y-axis
30 c (eg every 2.0 degrees latitude)
31 c
32 c OPTIONAL KEYWORDS:
33 c
34 c des=char -- The descriptor file to describe the format and the
35 c fields of each record of the input data
36 c (the default is the previous pipe).
37 c
38 c EXAMPLE:
39 c
40 c pbintag des=irtm_20.des x=lon y=lat xrange=0.0,10.0 yrange=0.0,10.0 \
41 c nbinx=120 nbiny=120 < irtm_20.data
42 c
