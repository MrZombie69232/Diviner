169  subroutine xyz2latlon(x,y,z,r0, dlat,dlon,htri)
170  real*4 x,y,z,dlat,dlon,r,r0,xi,yi,zi,htri
171  real*4 lat,lon,pi,r1
172  pi=dacos(-1.d0)
173  r=sqrt(x**2+y**2+z**2)
174 c find where x,y,z intersect sphere
175  xi=(r0/r)*x
176  yi=(r0/r)*y
177  zi=(r0/r)*z
178 c find latitude
179  lat=asin(zi/r0)
180  dlat=lat*180.d0/pi
181 c find projection onto equator
182  r1=sqrt(x**2+y**2)
183  lon=acos(x/r1)
184  if (y.lt.0.d0) lon=-lon
185  dlon=lon*180.d0/pi
186  htri=r-r0
187  return
188  end

190  subroutine ctricna(t, tc,tn,tarea)
191 c calculates triangle centers (tc),normals (tn) and triangele areas (tarea)
192  real*4 t(3,3),tn(3),tarea,u(3),v(3),cross(3),tc(3)
193 c calculate triangle centriod
194  tc(1)=(t(1,1)+t(1,2)+t(1,3))/3.
195  tc(2)=(t(2,1)+t(2,2)+t(2,3))/3.
196  tc(3)=(t(3,1)+t(3,2)+t(3,3))/3.
197 c set u and v, the difference vectors between the verticies
198  do i=1,3
199  u(i)=t(i,2)-t(i,1)
200  v(i)=t(i,3)-t(i,1)
201  enddo
202 c write (*,*) 'u= ',u
203 c write (*,*) 'v= ',v
204 
205 c calculate u cross v, which is tn
206  tn(1)=u(2)*v(3)-u(3)*v(2)
207  tn(2)=u(3)*v(1)-u(1)*v(3)
208  tn(3)=u(1)*v(2)-u(2)*v(1)
209 c calculate triangle area, which is 0.5*uXv
210  crossmag=sqrt(tn(1)**2+tn(2)**2+tn(3)**2)
211  tarea=0.5*crossmag
212 c normalize triange unit vectors to 1
213  tn(1)=tn(1)/crossmag
214  tn(2)=tn(2)/crossmag
215  tn(3)=tn(3)/crossmag
216  return
