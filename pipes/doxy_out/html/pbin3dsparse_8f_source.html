2 c PBIN3D: -- A three-dimensional binning program that computes
3 c the count, average, sigma, min and max for each bin.
4 c
5 c
6 c Required Keywords:

11 c range of each bin.
12 c z=char -- The third independent variable that defines the
13 c range of each bin.
14 c t=char -- The dependent variable that is dropped into
15 c the bins.
16 c
17 c
18 c xrange=double,double -- The range of values you want to
19 c bin, where the first is the minimum and the
20 c second is the maximum.
21 c
22 c deltax=double -- The increment of the bins.
23 c
24 c yrange=double,double -- The range of values you want to
25 c bin, where the first is the minimum and the
26 c second is the maximum.
27 c
28 c deltay=double -- The increment of the bins.
29 c
30 c zrange=double,double -- The range of values you want to
31 c bin, where the first is the minimum and the
32 c second is the maximum.
33 c
34 c deltaz=double -- The increment of the bins.
35 c
36 c Optional Keywords:
37 c
38 c des=char -- Descriptor file (default is previous pipe).
39 c
40 c sparse -- ounly output if count in all Z bins are > 0
41 c
42 c table -- outputs in tabular output with z avg values across the page
43 c
44 c exclude -- outputs this value if the count in a bin is zero
45 
46 c
47 c Output:
48 c
49 c Ascii file containing bin center values and results, with zeros for empty bins:
50 c
51 c xcenter,ycenter,zcenter,taverage,tcount,tsigma,tmin,tmax
